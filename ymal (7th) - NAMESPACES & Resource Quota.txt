  name: resources

  containers:
  - name: resource
    image: centos
    command: ["/bin/bash", "-c", "while true; do echo pod-resources; sleep 5 ; done"]
    resources:                                          
      requests:
        memory: "64Mi"
        cpu: "100m"
      limits:
        memory: "128Mi"
        cpu: "200m"

   name: myquota

  hard:
    limits.cpu: "400m"
    limits.memory: "400Mi"
    requests.cpu: "200m"
    requests.memory: "200Mi"
